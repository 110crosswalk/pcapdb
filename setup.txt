#!/bin/sh

POSTGRES_VERS=-9.4
PG_SERVICE=postgresql${POSTGRES_VERS}

# 1. Install packages listed in sys_requirements.txt

# 2. Setup postgres
if ! [ -e /var/lib/pgsql${POSTGRES_VERS}/data ]; then 
    service ${PG_SERVICE} initdb
fi

# The default postgresql configuration is fine. We'll only be connecting
# through localhost.
service ${PG_SERVICE} start
